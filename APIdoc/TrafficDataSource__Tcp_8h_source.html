133  /* Password lifecycle
134  *
135  * - The method onReadyRead detects that the device requests password. It
136  * will store the current SSID in passwordRequest_SSID and set
137  * passwordRequest_Status to waitingForPassword.
138  *
139  * - If a password for the SSID is found in the database, the method
140  * sendPassword is called with that password. Otherwise, the signal
141  * passwordRequest is emitted, which will hopefully lead to lead to a
142  * user-provided password through sendPassword()
143  *
144  * - The method send password will store the password in
145  * passwordRequest_password, send the password to the device and set
146  * passwordRequest_Status to waitingForDevice.
147  *
148  * - When the connection is closed while passwordRequest_Status ==
149  * waitingForDevice, this means that the traffic data receiver has
150  * rejected the password. The password stored in passwordRequest_password
151  * for passwordRequest_SSID is removed from the password database,
152  * passwordRequest_Status is set to idle, the members passwordRequest_SSID
153  * and passwordRequest_password are cleared and an immediate reconnect is
154  * scheduled.
155  *
156  * - When the heartbeat is received while passwordRequest_Status ==
157  * waitingForDevice, this means that the traffic data receiver has
158  * accepted the password. The instance will then emit the
159  * passwordStorageRequest. The member passwordRequest_Status is set to
160  * idle, and the members passwordRequest_SSID and passwordRequest_password
161  * are cleared.
162  */
163  enum {
164  /* No password-related activity is pending */
165  idle,
166 
167  /* Waiting for password
168  *
169  * A password has been requested by the traffic data receiver.
170  *
171  * At this stage, the member passwordRequest_SSID contains the network
172  * name of the WiFi network that the device was connected to at the
173  * time of the request.
174  */
175  waitingForPassword,
176 
177  /* Waiting for device
178  *
179  * A password has been sent to the traffic data receiver, and this
180  * class is now waiting for the device to respond.
181  *
182  * At this stage, the member passwordRequest_SSID contains the network
183  * name of the WiFi network that the device was connected to at the
184  * time of the request. The member passwordRequest_password contains
185  * the password that has been sent to the traffic data receiver
186  */
187  waitingForDevice
188  } passwordRequest_Status {idle};
