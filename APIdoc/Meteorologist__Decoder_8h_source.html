112  std::string roundTo(float number, size_t digitsAfterDecimalPoint)
113  {
114  static const char decimalPoint = '.';
115  std::string numStr = std::to_string(number);
116  const auto decimalPointPos = numStr.find(decimalPoint);
117  if (decimalPointPos == std::string::npos) return numStr;
118  const auto decimalsAfterPoint = numStr.length() - decimalPointPos;
119  if (decimalsAfterPoint > digitsAfterDecimalPoint) {
120  return numStr.substr(0, decimalPointPos + digitsAfterDecimalPoint + 1);
121  }
122  return numStr;
123  }

165  virtual QString visitTrendGroup(const TrendGroup & group, ReportPart reportPart, const std::string & rawString)
166  {
167  if (!group.isValid())
168  return tr("Invalid data");
169 
170  (void)reportPart; (void)rawString;
171  std::ostringstream result;
172  if (!group.isValid()) result << groupNotValidMessage << "\n";
173  switch (group.type()) {
174  case metaf::TrendGroup::Type::NOSIG:
175  result << "No significant weather changes expected";
176  break;
177 
178  case metaf::TrendGroup::Type::BECMG:
179  result << "Weather conditions are expected to gradually change ";
180  result << "as follows";
181  if (const auto timeFrom = group.timeFrom(); timeFrom) {
182  result << "\nFrom ";
183  result << explainMetafTime(*timeFrom).toStdString();
184  }
185  if (const auto timeUntil = group.timeUntil(); timeUntil) {
186  result << "\nUntil ";
187  result << explainMetafTime(*timeUntil).toStdString();
188  }
189  if (const auto timeAt = group.timeAt(); timeAt) {
190  result << "\nAt ";
191  result << explainMetafTime(*timeAt).toStdString();
192  }
193  break;
194 
195  case metaf::TrendGroup::Type::TEMPO:
196  result << "The following temporary weather conditions may arise ";
197  result << " for less than 60 minutes";
198  if (const auto timeFrom = group.timeFrom(); timeFrom) {
199  result << "\nFrom ";
200  result << explainMetafTime(*timeFrom).toStdString();
201  }
202  if (const auto timeUntil = group.timeUntil(); timeUntil) {
203  result << "\nUntil ";
204  result << explainMetafTime(*timeUntil).toStdString();
205  }
206  if (const auto timeAt = group.timeAt(); timeAt) {
207  result << "\nAt ";
208  result << explainMetafTime(*timeAt).toStdString();
209  }
210  if (group.probability() != metaf::TrendGroup::Probability::NONE) {
211  result << "\n";
212  result << probabilityToString(group.probability());
213  }
214  break;
215 
216  case metaf::TrendGroup::Type::INTER:
217  result << "The following temporary weather conditions may arise ";
218  result << "for less than 30 minutes";
219  if (const auto timeFrom = group.timeFrom(); timeFrom) {
220  result << "\nFrom ";
221  result << explainMetafTime(*timeFrom).toStdString();
222  }
223  if (const auto timeUntil = group.timeUntil(); timeUntil) {
224  result << "\nUntil ";
225  result << explainMetafTime(*timeUntil).toStdString();
226  }
227  if (const auto timeAt = group.timeAt(); timeAt) {
228  result << "\nAt ";
229  result << explainMetafTime(*timeAt).toStdString();
230  }
231  if (group.probability() != metaf::TrendGroup::Probability::NONE) {
232  result << "\n";
233  result << probabilityToString(group.probability());
234  }
235  break;
236 
237  case metaf::TrendGroup::Type::FROM:
238  result << "Weather conditions are expected to change rapidly from ";
239  result << explainMetafTime(*group.timeFrom()).toStdString();
240  result << "\nAll previous weather conditions are superseded by ";
241  result << "the following conditions";
242  break;
243 
244  case metaf::TrendGroup::Type::UNTIL:
245  result << "The following weather conditions expected to last until ";
246  result << explainMetafTime(*group.timeUntil()).toStdString();
247  break;
248 
249  case metaf::TrendGroup::Type::AT:
250  result << "The following weather conditions expected to occur at ";
251  result << explainMetafTime(*group.timeAt()).toStdString();
252  break;
253 
254  case metaf::TrendGroup::Type::TIME_SPAN:
255  result << "The following weather condition are expected within ";
256  result << "time span from ";
257  result << explainMetafTime(*group.timeFrom()).toStdString();
258  result << " until ";
259  result << explainMetafTime(*group.timeUntil()).toStdString();
260  if (group.probability() != metaf::TrendGroup::Probability::NONE) {
261  result << "\n";
262  result << probabilityToString(group.probability());
263  }
264  break;
265 
266  case metaf::TrendGroup::Type::PROB:
267  result << "The weather conditions probability is as follows\n";
268  result << probabilityToString(group.probability());
269  break;
270  }
271  return QString::fromStdString(result.str());
272  }

326  QString explainDistance(const metaf::Distance & distance) {
327  if (!distance.isReported()) return "not reported";
328  std::ostringstream result;
329  switch (distance.modifier()) {
330  case metaf::Distance::Modifier::NONE:
331  break;
332 
333  case metaf::Distance::Modifier::LESS_THAN:
334  result << "&lt;";
335  break;
336 
337  case metaf::Distance::Modifier::MORE_THAN:
338  result << "&gt;";
339  break;
340 
341  case metaf::Distance::Modifier::DISTANT:
342  result << "10 to 30 nautical miles ";
343  result << "(19 to 55 km, 12 to 35 statue miles)";
344  break;
345 
346  case metaf::Distance::Modifier::VICINITY:
347  result << "5 to 10 nautical miles ";
348  result << "(9 to 19 km, 6 to 12 statue miles)";
349  break;
350  }
351 
352  if (!distance.isValue())
353  return QString::fromStdString(result.str());
354 
355  if (distance.unit() == metaf::Distance::Unit::STATUTE_MILES) {
356  const auto d = distance.miles();
357  if (!d.has_value()) return "[unable to get distance value in miles]";
358  const auto integer = std::get<unsigned int>(d.value());
359  const auto fraction =
360  std::get<metaf::Distance::MilesFraction>(d.value());
361  if (integer || fraction == metaf::Distance::MilesFraction::NONE)
362  result << integer;
363  if (integer && fraction != metaf::Distance::MilesFraction::NONE)
364  result << " ";
365  if (fraction != metaf::Distance::MilesFraction::NONE)
366  result << distanceMilesFractionToString(fraction);
367  } else {
368  const auto d = distance.toUnit(distance.unit());
369  if (!d.has_value())
370  return "[unable to get distance's floating-point value]";
371  result << static_cast<int>(*d);
372  }
373  result << " " << distanceUnitToString(distance.unit());
374  result << " (";
375  if (distance.unit() != metaf::Distance::Unit::METERS) {
376  if (const auto d = distance.toUnit(metaf::Distance::Unit::METERS);
377  d.has_value())
378  {
379  result << static_cast<int>(*d);
380  result << " ";
381  result << distanceUnitToString(metaf::Distance::Unit::METERS);
382  } else {
383  result << "[unable to convert distance to meters]";
384  }
385  result << " / ";
386  }
387  if (distance.unit() != metaf::Distance::Unit::STATUTE_MILES) {
388  if (const auto d = distance.miles(); d.has_value()) {
389  if (!d.has_value())
390  return "[unable to get distance value in miles]";
391  const auto integer = std::get<unsigned int>(d.value());
392  const auto fraction =
393  std::get<metaf::Distance::MilesFraction>(d.value());
394  if (integer || fraction == metaf::Distance::MilesFraction::NONE)
395  result << integer;
396  if (integer && fraction != metaf::Distance::MilesFraction::NONE)
397  result << " ";
398  if (fraction != metaf::Distance::MilesFraction::NONE)
399  result << distanceMilesFractionToString(fraction);
400  result << " ";
401  result << distanceUnitToString(metaf::Distance::Unit::STATUTE_MILES);
402  } else {
403  result << "[unable to convert distance to statute miles]";
404  }
405  if (distance.unit() != metaf::Distance::Unit::FEET) result << " / ";
406  }
407  if (distance.unit() != metaf::Distance::Unit::FEET) {
408  if (const auto d = distance.toUnit(metaf::Distance::Unit::FEET);
409  d.has_value())
410  {
411  result << static_cast<int>(*d);
412  result << " " << distanceUnitToString(metaf::Distance::Unit::FEET);
413  } else {
414  result << "[unable to convert distance to feet]";
415  }
416  }
417  result << ")";
418  return QString::fromStdString(result.str());
419  }
